import org.springframework.web.client.RestTemplate;
import org.springframework.web.util.UriComponentsBuilder;
import lombok.RequiredArgsConstructor;

@Service
@RequiredArgsConstructor
public class KakaoPlaceService {
    private final RestTemplate restTemplate;

    @Value("${kakao.api.key}")
    private String apiKey;

    private static final String API_URL = "https://dapi.kakao.com/v2/local/search/keyword.json";

    public List<PlaceDto> searchPlaces(KakaoPlaceSearchParams params) {
        HttpHeaders headers = new HttpHeaders();
        headers.set("Authorization", "KakaoAK " + apiKey);

        UriComponentsBuilder builder = UriComponentsBuilder.fromHttpUrl(API_URL)
            .queryParam("query", params.getQuery())
            .queryParam("size", params.getSize() != null ? params.getSize() : 15);

        if (params.getCategoryGroupCode() != null) {
            builder.queryParam("category_group_code", params.getCategoryGroupCode());
        }
        if (params.getX() != null) {
            builder.queryParam("x", params.getX());
        }
        if (params.getY() != null) {
            builder.queryParam("y", params.getY());
        }

        try {
            HttpEntity<?> entity = new HttpEntity<>(headers);
            ResponseEntity<KakaoApiResponse> response = restTemplate.exchange(
                builder.toUriString(),
                HttpMethod.GET,
                entity,
                KakaoApiResponse.class
            );

            return response.getBody().getDocuments().stream()
                .map(place -> PlaceDto.builder()
                    .kakaoId(place.getId())
                    .name(place.getPlaceName())
                    .category(place.getCategoryName())
                    .address(place.getRoadAddressName() != null ?
                        place.getRoadAddressName() : place.getAddressName())
                    .phone(place.getPhone())
                    .latitude(Double.parseDouble(place.getY()))
                    .longitude(Double.parseDouble(place.getX()))
                    .placeUrl(place.getPlaceUrl())
                    .distance(place.getDistance())
                    .build())
                .collect(Collectors.toList());

        } catch (HttpClientErrorException e) {
            // Rate limit ì²˜ë¦¬
            if (e.getStatusCode().value() == 429) {
                throw new TooManyRequestsException("Kakao API rate limit exceeded");
            }
            throw e;
        }
    }

    public PlaceDetailDto getPlaceDetail(String placeId) {
        // ì¹´í…Œê³ ë¦¬ë³„ ìƒì„¸ ì •ë³´ ì¡°íšŒ

            "https://place.map.kakao.com/main/v/" + placeId,
            HttpMethod.GET,
            entity,
            KakaoPlaceDetailResponse.class
        );

        KakaoPlaceDetailResponse data = response.getBody();
        return PlaceDetailDto.builder()
            .businessHours(data.getBasicInfo() != null ? data.getBasicInfo().getOpenHour() : null)

function calculateCost(promptTokens: number, completionTokens: number, model: string): number {

      prompt: 0.01 / 1000,      // $0.01 per 1K tokens

      prompt: 0.0005 / 1000,
      completion: 0.0015 / 1000,
